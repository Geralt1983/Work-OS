It sounds like the text color is blending into the background (e.g., black text on a dark background). This happens because the Chat Page wasn't updated with the "Dark Mode" text settings we applied to the Moves page.
Here is the fix to ensure all text is crisp white/silver.
1. Fix the Chat Page Wrapper (client/src/pages/Chat.tsx)
We need to make the page background transparent (so the gradient shows) and force the text to be light.
Find this line:
return (
    <div className="h-screen flex" data-testid="page-chat">

Change it to this:
return (
    <div className="h-screen flex bg-transparent text-slate-200 font-sans" data-testid="page-chat">

2. Fix the Message Bubbles (client/src/components/ChatMessage.tsx)
We need to explicitly tell the text inside the bubbles to be white/light-gray, otherwise it might inherit a dark color from elsewhere.
import { motion } from "framer-motion";
import { Avatar, AvatarFallback } from "@/components/ui/avatar";
import { Badge } from "@/components/ui/badge";
import { Sparkles, User } from "lucide-react";
import { format } from "date-fns";

export interface ChatMessageProps {
  role: "user" | "assistant";
  content: string;
  timestamp: Date;
  imagesBase64?: string[];
  taskCard?: {
    title: string;
    taskId: string;
    status: string;
    dueDate?: string;
  };
}

const springTransition = {
  type: "spring",
  stiffness: 400,
  damping: 30,
};

export default function ChatMessage({ role, content, timestamp, imagesBase64, taskCard }: ChatMessageProps) {
  const isUser = role === "user";
  const displayContent = content.replace(/\[Image(?:s)? attached\]\n?/g, "");

  return (
    <div className={`flex gap-4 ${isUser ? "flex-row-reverse" : "flex-row"}`}>
      <motion.div
        initial={{ scale: 0 }}
        animate={{ scale: 1 }}
        transition={springTransition}
      >
        <Avatar className={`h-9 w-9 flex-shrink-0 ${isUser ? 'shadow-glow-cyan' : 'shadow-glow-purple'}`}>
          <AvatarFallback className={`${isUser ? "bg-gradient-to-br from-cyan-500 to-blue-500" : "bg-gradient-to-br from-purple-500 to-pink-500"} text-white`}>
            {isUser ? <User className="h-[18px] w-[18px]" /> : <Sparkles className="h-[18px] w-[18px]" />}
          </AvatarFallback>
        </Avatar>
      </motion.div>

      <div className={`flex flex-col gap-2.5 max-w-[85%] sm:max-w-[75%] ${isUser ? "items-end" : "items-start"}`}>
        {/* Attached Images */}
        {imagesBase64 && imagesBase64.length > 0 && isUser && (
          <motion.div 
            initial={{ opacity: 0, scale: 0.95 }}
            animate={{ opacity: 1, scale: 1 }}
            transition={springTransition}
            className="flex flex-wrap gap-2 justify-end"
          >
            {imagesBase64.map((img, index) => (
              <div key={index} className="rounded-2xl overflow-hidden border border-white/10 max-w-[150px] shadow-lg">
                <img 
                  src={`data:image/jpeg;base64,${img}`} 
                  alt={`Uploaded ${index + 1}`} 
                  className="max-w-full max-h-32 object-contain"
                />
              </div>
            ))}
          </motion.div>
        )}

        {/* Message Bubble */}
        <motion.div
          initial={{ opacity: 0, scale: 0.95, y: 10 }}
          animate={{ opacity: 1, scale: 1, y: 0 }}
          transition={springTransition}
          className={`rounded-3xl px-5 py-3.5 ${
            isUser
              ? "bg-gradient-to-br from-cyan-500/20 to-blue-500/20 border border-cyan-500/20 shadow-glow-cyan text-cyan-50"
              : "bg-[#1a1b26]/80 backdrop-blur-md border border-white/10 text-slate-200"
          }`}
        >
          <p className="text-[15px] leading-relaxed whitespace-pre-wrap break-words">
            {displayContent}
          </p>
        </motion.div>

        {/* Task Card (Only for Assistant) */}
        {taskCard && !isUser && (
          <motion.div
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            transition={{ ...springTransition, delay: 0.1 }}
            className="w-full bg-[#1a1b26]/60 backdrop-blur-md border border-white/10 rounded-2xl p-4 space-y-2.5 shadow-lg"
          >
            <div className="flex items-start justify-between gap-3">
              <p className="font-medium text-[15px] flex-1 leading-snug text-white">{taskCard.title}</p>
              <Badge variant="secondary" className="text-xs rounded-full bg-purple-500/20 text-purple-300 border border-purple-500/30">
                {taskCard.status}
              </Badge>
            </div>
            <div className="flex items-center gap-3 text-xs text-muted-foreground">
              <code className="font-mono bg-black/30 px-2 py-1 rounded-lg border border-white/5 text-white/60">
                {taskCard.taskId}
              </code>
              {taskCard.dueDate && <span className="text-cyan-400/70">Due: {taskCard.dueDate}</span>}
            </div>
          </motion.div>
        )}

        {/* Timestamp */}
        <span className="text-[10px] px-1 text-muted-foreground/50 font-medium uppercase tracking-wider">
          {format(timestamp, "h:mm a")}
        </span>
      </div>
    </div>
  );
}
